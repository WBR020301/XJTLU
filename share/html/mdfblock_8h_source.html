239  if (count != 1) {

246  if (count != 1) {

250  }

301 template <typename T>
302 void MdfBlock::WriteLink4List(std::FILE *file,
303  std::vector<std::unique_ptr<T>> &block_list,
304  size_t link_index, UpdateOption update_option) {
305  for (size_t index = 0; index < block_list.size(); ++index) {
306  auto &block = block_list[index];
307  //only update last_block.
308  const bool last_block = index + 1 >= block_list.size();
309  if (!block) {
310  continue;
311  }
312  bool need_update = false;
313  if (block->FilePosition() <= 0 ||
314  update_option == UpdateOption::DoUpdateAllBlocks) {
315  need_update = true;
316  }
317 
318  if (block->FilePosition() > 0 && last_block &&
319  update_option == UpdateOption::DoUpdateOnlyLastBlock) {
320  need_update = true;
321  }
322 
323  if (!need_update) {
324  continue;
325  }
326  block->Write(file);
327  if (index == 0) {
328  UpdateLink(file, link_index, block->FilePosition());
329  } else {
330  auto &prev = block_list[index - 1];
331  if (prev) {

333  }
334  }
