88  const auto block_size = BlockSize();
89  if (data_list_.empty() || block_size == 0) {
90  return;
91  }
92 
93  // Set up a temporary record buffer. It will be used as temporary
94  // buffer. It solves the RD vs RV/RI blocks data storage.
95  std::vector<uint8_t> record_buffer(block_size, 0);
96 
97  // Check if the data_list_ buffer is format as an RD block
98  // or as RV/RI blocks. The main idea is to copy to the
99  // temporary buffer, so it looks like an "RD" stored record.
100  if (IsRdBlock()) {
101  if (const auto mean_index = static_cast<int>(sample * 3 * block_size);
102  static_cast<size_t>(mean_index + block_size) <= data_list_.size()) {
103  std::copy_n(data_list_.cbegin() + mean_index, block_size, record_buffer.begin() );
104  }
105  value.MeanValid = channel.GetChannelValue(record_buffer, value.MeanValue, array_index);
106 
107  if (const auto min_index = static_cast<int>((sample * 3 * block_size) + block_size);
108  static_cast<size_t>(min_index + block_size) <= data_list_.size()) {
109  std::copy_n(data_list_.cbegin() + min_index, block_size, record_buffer.begin() );
110  }
111  value.MinValid = channel.GetChannelValue(record_buffer, value.MinValue, array_index);
112 
113  if (const auto max_index = static_cast<int>((sample * 3 * block_size) + (2 * block_size));
114  static_cast<size_t>(max_index + block_size) <= data_list_.size()) {
115  std::copy_n(data_list_.cbegin() + max_index, block_size, record_buffer.begin() );
116  }
117  value.MaxValid = channel.GetChannelValue(record_buffer, value.MaxValue, array_index);
118  } else {
119  const auto block_data_size = BlockDataSize();
120  const auto block_invalid_size = BlockInvalidSize();
121  const auto invalid_index = NofSamples() * 3 * block_data_size;

127  }

130  std::copy_n(data_list_.cbegin() + mean_invalid_index, block_invalid_size,
