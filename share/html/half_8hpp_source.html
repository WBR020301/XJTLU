716 inline void raise(int HALF_UNUSED_NOERR(flags),
717  bool HALF_UNUSED_NOERR(cond) = true) {
718 #if HALF_ERRHANDLING
719  if (!cond) return;
720 #if HALF_ERRHANDLING_FLAGS
721  errflags() |= flags;
722 #endif
723 #if HALF_ERRHANDLING_ERRNO
724  if (flags & FE_INVALID)
725  errno = EDOM;
726  else if (flags & (FE_DIVBYZERO | FE_OVERFLOW | FE_UNDERFLOW))
727  errno = ERANGE;
728 #endif
729 #if HALF_ERRHANDLING_FENV && HALF_ENABLE_CPP11_CFENV
730  std::feraiseexcept(flags);
731 #endif
732 #ifdef HALF_ERRHANDLING_THROW_INVALID
733  if (flags & FE_INVALID)
734  throw std::domain_error(HALF_ERRHANDLING_THROW_INVALID);
735 #endif
736 #ifdef HALF_ERRHANDLING_THROW_DIVBYZERO
737  if (flags & FE_DIVBYZERO)
738  throw std::domain_error(HALF_ERRHANDLING_THROW_DIVBYZERO);
739 #endif
740 #ifdef HALF_ERRHANDLING_THROW_OVERFLOW
741  if (flags & FE_OVERFLOW)
742  throw std::overflow_error(HALF_ERRHANDLING_THROW_OVERFLOW);
743 #endif
744 #ifdef HALF_ERRHANDLING_THROW_UNDERFLOW
745  if (flags & FE_UNDERFLOW)
746  throw std::underflow_error(HALF_ERRHANDLING_THROW_UNDERFLOW);
747 #endif
748 #ifdef HALF_ERRHANDLING_THROW_INEXACT
749  if (flags & FE_INEXACT)
750  throw std::range_error(HALF_ERRHANDLING_THROW_INEXACT);
751 #endif
752 #if HALF_ERRHANDLING_UNDERFLOW_TO_INEXACT
753  if ((flags & FE_UNDERFLOW) && !(flags & FE_INEXACT)) raise(FE_INEXACT);
754 #endif
755 #if HALF_ERRHANDLING_OVERFLOW_TO_INEXACT
756  if ((flags & FE_OVERFLOW) && !(flags & FE_INEXACT)) raise(FE_INEXACT);
757 #endif
758 #endif
759 }
